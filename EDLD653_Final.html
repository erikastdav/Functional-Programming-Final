API calls to chess.com

Chess.com is an internet chess server. Live online chess can be played against other users in daily, rapid, blitz or bullet time controls, with a number of chess variants also available. It is one of the largest chess platforms in the world. In this part of the assignment, we will interact with their API to scrap some data. Below is some information to start with.

You can get a list of the usernames for 10,000 players from the USA using the R code in the chunk below. Due to performance reasons, it doesn’t return the list of all players. Note that the returned list of usernames may change from day to day so you may not get the exact same list as others. (More information about chess.com API: https://www.chess.com/news/view/published-data-api)

In the code chunk below, player_list is a character vector of length 10,000 where each element is a username.

#profile
get_profile <- function(username) {
  profile_response <- GET(paste0("https://api.chess.com/pub/player/", username))
  profile <- fromJSON(rawToChar(profile_response$content))
  
#stats
  stats_response <- GET(paste0("https://api.chess.com/pub/player/", username, "/stats"))
  stats <- fromJSON(rawToChar(stats_response$content))
  
#extractubg the things listed in the question
  data.frame(
    username        = username,
    name            = ifelse(is.null(profile$name),     NA, profile$name),
    followers       = ifelse(is.null(profile$followers), NA, profile$followers),
    location        = ifelse(is.null(profile$location),  NA, profile$location),
    last_online     = as.POSIXct(profile$last_online, origin = "1970-01-01", tz = "UTC"),
    
    rapid_last_rating  = ifelse(is.null(stats$chess_rapid$last$rating),  NA, stats$chess_rapid$last$rating),
    rapid_best_rating  = ifelse(is.null(stats$chess_rapid$best$rating),  NA, stats$chess_rapid$best$rating),
    rapid_record_win   = ifelse(is.null(stats$chess_rapid$record$win),   NA, stats$chess_rapid$record$win),
    rapid_record_loss  = ifelse(is.null(stats$chess_rapid$record$loss),  NA, stats$chess_rapid$record$loss),
    rapid_record_draw  = ifelse(is.null(stats$chess_rapid$record$draw),  NA, stats$chess_rapid$record$draw),
    
    stringsAsFactors = FALSE
  )
}

estimate_reliability <- function(df, subject_col) {
  
  df$score <- df[[subject_col]]
  
#numerator
  
  lme_model <- suppressWarnings(
    lmer(score ~ 1 + wave + (1 + wave | sid), REML = FALSE, data = df)
  )
  vc       <- as.data.frame(VarCorr(lme_model))
  true_var <- vc$vcov[vc$var1 == "wave" & is.na(vc$var2)]

#denominator
  
  student_slopes <- sapply(split(df, df$sid), function(student) {
    coef(lm(score ~ 1 + wave, data = student))[["wave"]]
  })
  observed_var <- var(student_slopes)
  
  data.frame(
    subject      = subject_col,
    true_var     = round(true_var,     4),
    observed_var = round(observed_var, 4),
    reliability  = round(true_var / observed_var, 4)
  )
}

#function 1: bootstrap n students w replacement (all time pts) 

bootstrap_sample <- function(df, id_var, n_bootstraps) {
  unique_ids  <- unique(df[[id_var]])
  sampled_ids <- sample(unique_ids, size = length(unique_ids), replace = TRUE)
  
  boot_list <- lapply(seq_along(sampled_ids), function(i) {
    student_data     <- df[df[[id_var]] == sampled_ids[i], ]
    student_data$new_id <- paste0(sampled_ids[i], "_", i) 
    student_data
  })
  
  do.call(rbind, boot_list)
}

#function 2: slope reliability; take df, id, time, and outcome as the arguments 

calc_reliability <- function(df, id_var, time_var, outcome_var) {

  lme_formula <- as.formula(paste(outcome_var, "~ 1 +", time_var, "+ (1 +", time_var, "|", id_var, ")"))
  ols_formula <- as.formula(paste(outcome_var, "~ 1 +", time_var))
  
  # numerator: random slope variance from lmer
  model    <- suppressWarnings(
    lmer(lme_formula, REML = FALSE, data = df)
  )
  vc       <- as.data.frame(VarCorr(model))
  true_var <- vc$vcov[vc$var1 == time_var & is.na(vc$var2)]
  
  # denominator: variance of OLS slopes across students
  student_slopes <- sapply(split(df, df[[id_var]]), function(student) {
    coef(lm(ols_formula, data = student))[[time_var]]
  })
  observed_var <- var(student_slopes)
  
  return(true_var / observed_var)
}
